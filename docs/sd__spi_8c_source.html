811 dword crc (byte *data, int count, dword poly)
812 {
813  const int BITS_DWORD = 8 * sizeof (dword);
814  const int bits = count * 8 * sizeof (byte);
815 
816  int i, count_dword; /* other */
817  int start_bit, cur_length; /* data preparation */
818  int digits; /* division */
819 
820  dword *array;
821  dword crc;
822 
823  /* get maximum power of 2 in polynomial */
824  i = BITS_DWORD - 1;
825  while (GETBIT (poly, i) == 0)
826  i--;
827 
828  const int POLY_BITS = i + 1;
829  const int TOTAL_BITS = bits + i;
830 
831  /* prepare dword array for M(x) * x^k */
832  count_dword = TOTAL_BITS / BITS_DWORD;
833  if (TOTAL_BITS % BITS_DWORD)
834  count_dword += 1;
835 
836  array = (dword *)malloc (sizeof (dword) * count_dword);
837  memset (array, 0, sizeof (dword) * count_dword); /* init to 0 */
838 
839  /* copy bytes in dword array */
840  start_bit = 0;
841  i = count_dword - 1;
842  while (start_bit < bits)
843  {
844  if ((cur_length = (bits - start_bit)) >= BITS_DWORD)
845  cur_length = BITS_DWORD;
846 
847  array[i] = get_bits (data, count, start_bit, cur_length);
848  start_bit += cur_length;
849  i--;
850  }
851 
852  /* do M(x) * x^k */
853  for (i = 0; i < (POLY_BITS - 1); i++)
854  l_shift (array, count_dword);
855 
856  /* prepare for division */
857 
858  /* align polynomial */
859  while (GETBIT (poly, (BITS_DWORD - 1)) == 0)
860  poly <<= 1;
861 
862 
863  /* init to number of total bits in dword representation */
864  digits = BITS_DWORD * count_dword;
865  do
866  {
867  /* subtract polynomial if possible */
868  if (GETBIT (array[0], (BITS_DWORD - 1)) != 0)
869  array[0] ^= poly;
870 
871  /* align number */
872  l_shift (array, count_dword);
873  digits--;
874  } while (digits >= POLY_BITS); /* exit if message cannot not be divided anymore */
875 
876  crc = array[0] >> (BITS_DWORD - (POLY_BITS - 1));
877  free (array);
878 
879  return crc;
880 }

888 void l_shift (dword *data, int count)
889 {
890  int i;
891 
892  /* bitwise right shift array of 1 position */
893  for (i = 0; i < count - 1; i++)
894  {
895  data[i] <<= 1;
896  data[i] |= (data[i + 1] >> ((8 * sizeof (dword)) - 1));
897  }
898 
899  data[count - 1] <<= 1;
900 
901  return;
902 }
